function calculoDescuento (precio, porcentajeDesc){ //Creación de la función
    const descuento = (precio * porcentajeDesc) / 100;
    const precioConDescuento = precio - descuento;

    return precioConDescuento;
}

const precioOriginal = 1500;
const porcentajeDesc = 35;

const precioFinal = calculoDescuento(precioOriginal, porcentajeDesc);

console.log("Precio original: $" + precioOriginal);
console.log("Descuento: " + porcentajeDesc + "%");
console.log("Precio final: $" + precioFinal);
